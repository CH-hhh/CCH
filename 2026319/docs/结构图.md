webssh/
├── Makefile                # 负责控制项目构建的配置文件
├── webssh_server           # 编译生成的核心可执行文件
├── webssh.db               # (或 webssh_log.db) SQLite 运行后的数据库文件
│
├── src/                    # C 语言源代码目录 (所有源文件文件)
│   ├── server.c            # 网络连入主入口、服务监听与路由分发
│   ├── file_io.c           # 执行系统命令、处理管道流式通信与目录隔离
│   ├── db_helper.c         # SQLite 第三方数据库调用与预编译插入日志
│   └── cmd_validator.c     # 命令分类器、反转义及编辑器降维存取
│
├── include/                # 暴露的 C 语言头文件目录
│   ├── file_io.h           # 相关 IO 函数方法声明
│   ├── db_helper.h         # 数据库相关方法声明
│   └── cmd_validator.h     # 验证、存取命令声明
│
├── web/                    # 前端视图资源目录
│   └── index.html          # WebSSH 终极网页端视窗 (涵盖流式拉取、显示等 JS 逻辑)
│
└── docs/                   # 自动文档与报告归档目录
    ├── 项目详尽分析报告.md
    ├── webssh系统能力培养报告.md
    └── webssh_深度技术白皮书_V2.md
